```jsx
import { ColorPicker, ThemeSdk, ThemeScene, Notify } from 'zent';
import { useState } from 'react';

const { generateTheme, applyTheme } = ThemeSdk;

const colors = [
  '#ed6a18',
  '#ed9f18',
  '#edd418',
  '#bad415',
  '#5bd415',
  '#15d48e',
  '#15bad4',
  '#155bd4',
  '#412ad4',
  '#8e15d4',
  '#d415ba',
  '#d42f15',
];

const specificScene = ThemeScene.DefaultHoverBackgroundColor;

const defaultTheme = generateTheme({ colors: [{ baseColor: '#155bd4', scene: specificScene }] });

const ThemeSdkComponent = () => {
  const [color, setColor] = useState('#155bd4');

  const [defaultHoverColor, setDefaultHoverColor] = useState(defaultTheme?.colors[0]);

  const onChangeComplete = hex => {
    const theme = generateTheme({ colors: [{ baseColor: hex, scene: specificScene }] });
    applyTheme(theme);

    setColor(hex);
    setDefaultHoverColor(theme.colors[0]);

    Notify.success('{i18n.prompt}');
  };

  return (
    <>
      <ColorPicker color={color} type="simple" onChange={onChangeComplete} presetColors={colors} />
      <span className="deme-theme-base-color">{color}</span>
      <div className="demo-theme-colors">
         <div className="demo-theme-item">
            <div
              className="demo-theme-color"
              style={{ background: defaultHoverColor.color }}
            >
              {defaultHoverColor.color}
            </div>
          </div>
      </div>
    </>
  );
}

ReactDOM.render(<ThemeSdkComponent />, mountNode);
```
