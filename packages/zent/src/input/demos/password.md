```jsx
import { Password } from 'zent';

class EventTest extends React.Component {
	state = {
		value: '',
	};

	onChange = e => {
		this.setState({ value: e.target.value });
	};

	render() {
		return (
			<div>
				<Password />
				<Password
					onChange={this.onChange}
					value={this.state.value}
					showClear
				/>
			</div>
		);
	}
}

ReactDOM.render(
	<EventTest />,
	mountNode
);
```
