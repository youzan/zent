```jsx
import { Popover, Button } from 'zent';

ReactDOM.render(
	<Popover
		className="zent-doc-popover"
		position={Popover.Position.BottomLeft}
		cushion={5}
	>
		<Popover.Trigger.Click>
			<Button type="primary">{i18n.click}</Button>
		</Popover.Trigger.Click>
		<Popover.Content>
			<div>{i18n.msg1}</div>
			<div>{i18n.msg2}</div>
		</Popover.Content>
	</Popover>,
	mountNode
);
```
